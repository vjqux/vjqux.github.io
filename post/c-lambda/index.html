// 显示返回function对象的lambda和采用function对象作为其参数的lambda
int main() {
    using namespace std;
    auto addtwointegers = [](int x) -> function<int(int)> {
        return [=](int y) {return x + y；}；
    }

    auto higerorder = [](const function<int(int)>& f, int z) {
        return fj(z) * 2;
    }

    auto answer = higerorder (addtwointegers(7), 8);

    cout<<answer<<endl;
}
